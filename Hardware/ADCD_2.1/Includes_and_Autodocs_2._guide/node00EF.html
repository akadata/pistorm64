#define PI	  ((float) 3.141592653589793)
#endif

#endif

#define sqrt	SPSqrt

#define	sinh	SPSinh
#define cosh	SPCosh
#define tanh	SPTanh

float	 afp(),    dbf();		  /* Math conversion functions */
