{
    ULONG pp_Reserved[4];

    UWORD pp_Height;				/* Height in pixels */
    UWORD pp_Depth;				/* Depth */
    UWORD pp_YSize;				/* YSize */

    /* Color Table:  numEntries = (1 << pp_Depth) - 1 */

    /* Data follows */
};
