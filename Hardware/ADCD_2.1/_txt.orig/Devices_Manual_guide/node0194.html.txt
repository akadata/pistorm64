#define READ_BUFFER_SIZE 25 UBYTE ReadBuff[READ_BUFFER_SIZE]; UWORD ctr; if (SerialMP=CreatePort(0,0) ) { if (SerialIO=(struct IOExtSer *) CreateExtIO(SerialMP,sizeof(struct IOExtSer))) { if (OpenDevice(SERIALNAME,0L,(struct IORequest *)SerialIO,0) ) printf("%s did not open\n",SERIALNAME); else { /* Tell user what we are doing */ printf("\fLooking for Q, E, EOT or ETX\n"); /* Set EOF mode flag * Set the termination array * Send SDCMD_SETPARAMS to the serial device */ SerialIO->io_SerFlags |= SERF_EOFMODE; SerialIO->io_TermArray = Terminators; SerialIO->IOSer.io_Command = SDCMD_SETPARAMS; if (DoIO((struct IORequest *)SerialIO)) printf("Set Params failed "); /* Inform user of error */ else { SerialIO->IOSer.io_Length = READ_BUFFER_SIZE;
